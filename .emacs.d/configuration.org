** Theme

#+BEGIN_SRC emacs-lisp
 (use-package dbus
  :ensure nil
  :config
  (defun my-apply-system-theme (value)
    "Apply Emacs theme based on the freedesktop color-scheme value.
     1 = dark preference, 0 or 2 = light/no preference."
    ;; DBus can pass the value raw or wrapped in a list. This safely extracts it.
    (let ((theme-val (if (listp value) (car value) value)))
      (if (equal theme-val 1)
	  (progn
	    ;; Always disable the old theme before loading the new one
	    (disable-theme 'dichromacy)
	    (load-theme 'wombat t)
	    (message "System theme applied: Dark (wombat)"))
	(progn
	  (disable-theme 'wombat)
	  (load-theme 'dichromacy t)
	  (message "System theme applied: Light (dichromacy)")))))

  (defun my-dbus-theme-signal-handler (namespace key value)
    "Triggered when GNOME/Freedesktop changes the theme."
    (when (and (string-equal namespace "org.freedesktop.appearance")
	       (string-equal key "color-scheme"))
      (my-apply-system-theme (car value))))

  ;; 1. CHECK ON STARTUP
  ;; We use a synchronous call here so Emacs knows the theme immediately
  ;; before rendering the UI. Wrapped in condition-case so Emacs won't crash
  ;; if DBus isn't running.
  (condition-case nil
      (let ((current-theme (dbus-call-method
			    :session
			    "org.freedesktop.portal.Desktop"
			    "/org/freedesktop/portal/desktop"
			    "org.freedesktop.portal.Settings"
			    "Read"
			    "org.freedesktop.appearance"
			    "color-scheme")))
	;; `Read` returns a variant inside a list, so we pass the `car`
	(my-apply-system-theme (car current-theme)))
    (error (message "Warning: Could not read DBus theme at startup.")))

  ;; 2. LISTEN FOR CHANGES
  ;; Registers the background listener for when you toggle the system theme
  (dbus-register-signal
   :session
   "org.freedesktop.portal.Desktop"
   "/org/freedesktop/portal/desktop"
   "org.freedesktop.portal.Settings"
   "SettingChanged"
   #'my-dbus-theme-signal-handler))
#+End_SRC

  ;; The Eglot Gatekeeper
  (defun my-eglot-latex-gate (orig-fn &rest args)
    (if (and (derived-mode-p 'latex-mode 'TeX-mode 'LaTeX-mode)
	     (let ((start-pos (save-excursion (skip-syntax-backward "w_") (point))))
	       (not (memq (char-before start-pos) '(?\\ ?\{ ?\[)))))

  (add-hook 'eglot-managed-mode-hook
	    (lambda ()
	      (when (derived-mode-p 'latex-mode 'TeX-mode 'LaTeX-mode)
