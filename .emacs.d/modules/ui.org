* Theme and appearance

#+begin_src emacs-lisp
(my/use-package dbus
  :ensure nil
  :config
  (defun my-apply-system-theme (value)
    "Apply Emacs theme based on the freedesktop color-scheme VALUE.
1 means dark preference, 0 or 2 means light/no preference."
    (let ((theme-val (if (listp value) (car value) value)))
      (cond
       ((equal theme-val 1)
        (mapc #'disable-theme custom-enabled-themes)
        (load-theme 'wombat t)
        (message "System theme applied: Dark (wombat)"))
       (t
        (mapc #'disable-theme custom-enabled-themes)
        (load-theme 'dichromacy t)
        (message "System theme applied: Light (dichromacy)")))))

  (defun my-dbus-theme-signal-handler (namespace key value)
    "Triggered when the system theme changes."
    (when (and (string-equal namespace "org.freedesktop.appearance")
               (string-equal key "color-scheme"))
      (my-apply-system-theme value)))

  (condition-case nil
      (let ((current-theme
             (dbus-call-method
              :session
              "org.freedesktop.portal.Desktop"
              "/org/freedesktop/portal/desktop"
              "org.freedesktop.portal.Settings"
              "Read"
              "org.freedesktop.appearance"
              "color-scheme")))
        (my-apply-system-theme (car current-theme)))
    (error
     (load-theme 'wombat t)
     (message "Warning: Could not read DBus theme at startup. Falling back to wombat.")))

  (ignore-errors
    (dbus-register-signal
     :session
     "org.freedesktop.portal.Desktop"
     "/org/freedesktop/portal/desktop"
     "org.freedesktop.portal.Settings"
     "SettingChanged"
     #'my-dbus-theme-signal-handler)))

(my/use-package hydra
  :ensure nil
  :bind ("C-c w" . hydra-window/body)
  :config
  (defhydra hydra-window (:color red :hint nil)
    "
^Navigation^        ^Split/Delete^       ^Workspaces (Persp)^    ^Sizing^
^^^^^^^^--------------------------------------------------------------------------
_h_/_j_/_k_/_l_: move  _v_: vertical split  _s_: switch persp       _+_: taller
_w_: other window   _x_: horiz split     _n_: next persp         _-_: shorter
_0_-_9_: select #   _d_: delete window   _p_: prev persp         _>_: wider
_t_: treemacs       _o_: delete others   _c_: create persp       _<_: narrower
"
    ("h" windmove-left) ("j" windmove-down) ("k" windmove-up) ("l" windmove-right)
    ("w" other-window) ("1" winum-select-window-1) ("2" winum-select-window-2)
    ("3" winum-select-window-3) ("4" winum-select-window-4) ("5" winum-select-window-5)
    ("6" winum-select-window-6) ("7" winum-select-window-7) ("8" winum-select-window-8)
    ("9" winum-select-window-9) ("0" treemacs-select-window) ("t" treemacs)
    ("v" (lambda () (interactive) (split-window-right) (other-window 1)))
    ("x" (lambda () (interactive) (split-window-below) (other-window 1)))
    ("d" delete-window) ("o" delete-other-windows) ("s" persp-switch)
    ("n" persp-next) ("p" persp-prev) ("c" persp-create)
    ("+" enlarge-window) ("-" shrink-window)
    (">" enlarge-window-horizontal) ("<" shrink-window-horizontal)
    ("q" nil "exit" :color blue) ("C-g" nil "exit" :color blue)))
