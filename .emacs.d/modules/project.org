#+TITLE: Project and files
#+PROPERTY: header-args:emacs-lisp :tangle yes

* Projects

#+begin_src emacs-lisp
  (my/use-package project
    :bind (("C-c p f" . project-find-file)
           ("C-c p b" . project-switch-to-buffer)
           ("C-c p p" . project-switch-project)))

  (my/use-package treemacs
    :bind (("C-c t" . treemacs)
           ("M-0"   . treemacs-select-window))
    :config
    (setq treemacs-collapse-dirs 3
          treemacs-deferred-git-apply-delay 0.5
          treemacs-display-in-side-window t
          treemacs-file-follow-delay 0.2
          treemacs-follow-after-init t
          treemacs-expand-after-init t
          treemacs-hide-dot-git-directory t
          treemacs-indentation 2
          treemacs-position 'left
          treemacs-show-hidden-files t
          treemacs-width 35)
    (treemacs-follow-mode 1)
    (treemacs-filewatch-mode 1)
    (treemacs-fringe-indicator-mode 'always)
    (treemacs-git-mode 'deferred))

  (my/use-package treemacs-icons-dired
    :hook (dired-mode . treemacs-icons-dired-enable-once))

  (my/use-package treemacs-persp
    :after (treemacs perspective)
    :config (treemacs-set-scope-type 'Perspectives))

    (my/use-package dired-subtree
    :bind (:map dired-mode-map
                ("i" . dired-subtree-toggle)))

  (my/use-package dired-collapse
    :hook (dired-mode . dired-collapse-mode))

(my/use-package dired-filter
    :commands (dired-filter-group-mode dired-filter-group-local-set dired-filter-by-garbage)
    :hook (dired-mode . (lambda ()
                          (dired-filter-group-mode 1)
                          (dired-filter-group-local-set "default")
                          (dired-filter-by-garbage)))
    :custom
    (dired-garbage-files-regexp "\\(?:\\.\\(?:aux\\|bak\\|dvi\\|log\\|orig\\|rej\\|toc\\|out\\)\\)\\'")
    (dired-filter-group-saved-groups
      '(("default"
         ("Org"          (extension "org"))
         ("Executables"  (executable))
         ("Directories"  (directory))
         ("PDF"          (extension "pdf"))
         ("LaTeX"        (extension "tex" "bib"))
         ("Images"       (extension "png" "jpg" "jpeg"))
         ("Code"         (extension "py" "jl" "rs" "go" "c"))
         ("Archives"     (extension "zip" "rar" "gz" "bz2" "tar"))))))

  (my/use-package perspective
    :init
    (setq persp-mode-prefix-key (kbd "C-c M-p"))
    :config
    (persp-mode 1)
    (persp-turn-on-modestring))
#+end_src
