(my/use-package gnuplot)

(with-eval-after-load 'org
  (let ((available-langs '()))
    (dolist (lang '(C emacs-lisp gnuplot julia latex org python shell R))
      (if (locate-library (concat "ob-" (symbol-name lang)))
          (push (cons lang t) available-langs)
        (message "Warning: ob-%s not found, skipping optimization." lang)))
       (org-babel-do-load-languages 'org-babel-load-languages available-langs)))
