* Package bootstrap

#+begin_src emacs-lisp
(defun my/local-package-available-p (pkg)
  "Return non-nil if PKG is available locally."
  (or (locate-library (symbol-name pkg))
      (package-installed-p pkg)
      (featurep pkg)))

(defmacro my/use-package (name &rest args)
  "Prefer system packages, fallback to ELPA when missing."
  (declare (indent defun))
  `(use-package ,name
     ,@(unless (my/local-package-available-p name)
         '(:ensure t :demand t))
     ,@args))

(setq use-package-always-defer t)
(setq use-package-compute-statistics t)

(setq native-comp-async-report-warnings-errors nil)
#+end_src
