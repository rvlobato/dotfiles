(my/use-package gnus
  :defer t
  :bind (("C-c M" . gnus)) ; Shifted to uppercase M to resolve the Consult collision
  :config
  (setq gnus-select-method '(nnnil)
        gnus-secondary-select-methods
        '((nntp "news.gwene.org")
          (nnimap "gmail"
                  (nnimap-address "imap.gmail.com")
                  (nnimap-server-port "imaps")
                  (nnimap-stream ssl)
                  (nnmail-expiry-target "nnimap+gmail:[Gmail]/Trash")
                  (nnmail-expiry-wait immediate))))
  (setq smtpmail-smtp-server "smtp.gmail.com"
        smtpmail-smtp-service 587
        gnus-ignored-newsgroups "^to\\.\\|^[0-9. ]+\\( \\|$\\)")
  (defun my-message-mode-setup ()
    (setq fill-column 72)
    (turn-on-auto-fill))
  (add-hook 'message-mode-hook #'my-message-mode-setup)
  (add-hook 'gnus-group-mode-hook #'gnus-topic-mode)
  (setq mail-user-agent 'message-user-agent
        compose-mail-user-agent-warnings nil
        message-mail-user-agent nil
        mail-signature "Ronaldo V. Lobato\nrvlobato.com\n"
        message-signature "Ronaldo V. Lobato\nrvlobato.com\n"
        message-citation-line-format "%f [%Y-%m-%d, %R %z]:\n"
        message-citation-line-function 'message-insert-formatted-citation-line
        message-confirm-send nil
        message-kill-buffer-on-exit t
        message-wide-reply-confirm-recipients t
        message-default-charset 'utf-8
        gnus-gcc-mark-as-read t
        gnus-agent t
        gnus-novice-user nil)
  (setq gnus-check-new-newsgroups 'ask-server
        gnus-read-active-file 'some)
  (setq gnus-use-dribble-file nil
        gnus-use-cache 'use-as-much-cache-as-possible
        gnus-asynchronous t
        gnus-use-article-prefetch 15)
  (setq gnus-home-directory "~/Documents/gnus"
        nnfolder-directory "~/Documents/gnus/Mail/archive"
        message-directory "~/Documents/gnus/Mail"
        nndraft-directory "~/Documents/gnus/Drafts"
        gnus-cache-directory "~/Documents/gnus/cache")
  (setq mm-encrypt-option 'guided)
  (setq mml-secure-openpgp-encrypt-to-self t
        mml-secure-openpgp-sign-with-sender t
        mml-secure-smime-encrypt-to-self t
        mml-secure-smime-sign-with-sender t)
  
  ;; Secure the background execution engine dependencies safely
  (with-eval-after-load 'smtpmail
    (require 'smtpmail-async))
    
  (setq send-mail-function 'async-smtpmail-send-it
        message-send-mail-function 'async-smtpmail-send-it))
