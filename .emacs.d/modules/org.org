* Basic Org settings

#+begin_src emacs-lisp
  (my/use-package org
    :custom
    (org-directory "~/Documents/cloud/org/")
    (org-agenda-files '("~/Documents/cloud/org/agenda.org"
                        "~/Documents/cloud/org/notes.org"))
    (org-agenda-tags-column -10)
    (org-agenda-span 'day)
    (org-deadline-warning-days 14)
    (org-agenda-show-all-dates t)
    (org-agenda-skip-scheduled-if-done t)
    (org-agenda-skip-deadline-if-done t)
    (org-agenda-start-on-weekday nil)
    (org-startup-folded t)
    (org-hide-block-startup t)
    (org-list-allow-alphabetical t)
    (org-src-preserve-indentation t)
    (org-src-tab-acts-natively t)
    (org-fontify-quote-and-verse-blocks t)
    (org-return-follows-link t)
    (org-startup-with-inline-images t)
    (org-image-actual-width 500)
    (org-html-validation-link nil)
    (org-enforce-todo-dependencies t)
    (org-lowest-priority ?D)
    :config
    (global-set-key (kbd "C-c a") #'org-agenda)
    (global-set-key (kbd "C-c c") #'org-capture)

    (setq org-catch-invisible-edits 'smart
          org-hide-leading-stars t
          org-startup-indented t
          org-support-shift-select t
          org-src-fontify-natively t
          org-confirm-babel-evaluate nil
          org-export-babel-evaluate nil)

    (setq org-priority-faces
          '((?A :foreground "red" :weight bold)
            (?B . "orange")
            (?C . "yellow")
            (?D . "green")))

    (setq org-html-postamble "Created by: %a on %d. Last Updated: %C")
    (add-hook 'org-mode-hook 'which-function-mode)
    (add-hook 'org-babel-after-execute-hook 'org-display-inline-images)

    ;; Scientific Capture Templates
    (setq org-capture-templates
          '(("t" "Todo [General]" entry (file+headline "~/Documents/cloud/org/agenda.org" "Inbox")
             "* TODO %?\n  %i\n  %a")
            ("p" "Physics / ML Pipeline Task" entry (file+headline "~/Documents/cloud/org/agenda.org" "Research")
             "* TODO %? :pipeline:ml:\n  %i\n  %a")
            ("a" "CBPF Admin/Grant Task" entry (file+headline "~/Documents/cloud/org/agenda.org" "Admin")
             "* TODO %? :cbpf:admin:\n  SCHEDULED: %^t\n  %i\n  %a")
            ("l" "Literature to Read" entry (file+headline "~/Documents/cloud/org/notes.org" "Literature")
             "* TO-READ %? :arxiv:paper:\n  %i\n  %a"))))

  ;; Habit Module Activation
  (with-eval-after-load 'org
    (add-to-list 'org-modules 'org-habit t)
    (require 'org-habit))
#+end_src
